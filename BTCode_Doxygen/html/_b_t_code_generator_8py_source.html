49 def parse_input_line(line):
50  line_spt = line.split(":")
51  node_name = line_spt[0].strip(' ')
52  operator = None
53  if " and " in line_spt[1]:
54  operator = "sequence"
55  if " or " in line_spt[1]:
56  operator = "selector"
57  children = list()
58  names = line_spt[1].split(' ')
59  for j in range(0,len(names)):
60  curr_name = names[j].strip(' ').strip(',')
61  if len(curr_name) != 0:
62  if curr_name != "and" and curr_name != "or":
63  children.append(curr_name)
64  return operator, node_name, children

85

88  seq_list.append(node_name)
89  if node_name in tasks:
90  tasks.remove(node_name)
91

93  bt_graph.node(node_name, node_name, color='red')
94  sel_list.append(node_name)
95  if node_name in tasks:
96  tasks.remove(node_name)
97

100  bt_graph.node(children[i], children[i])
101  tasks.append(children[i])
